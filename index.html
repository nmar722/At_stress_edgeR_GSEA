1 EdgeR analysis of RNAseq data of Arabidopsis stress experiment

The RNAseq count data from transcription profiling experiment by high throughput sequencing of Arabidopsis plants in response to a combination of salt and heat stress have been downloaded from Expression Atlas website.

# Load edgeR
library(edgeR)

1.1 Load the data and create DGE list

# Load the count data
raw_counts <- read.delim("E-GEOD-72806-raw-counts.tsv", stringsAsFactors = FALSE)
# Preserve gene ID and gene name
gene_info <- raw_counts[, c("Gene.ID", "Gene.Name")]
# Extract count matrix (assuming counts start from column 3 onward)
count_matrix <- raw_counts[, 3:ncol(raw_counts)]
# Set rownames to Gene.ID
rownames(count_matrix) <- raw_counts$Gene.ID
# Load metadata
metadata <- read.delim("E-GEOD-72806-experiment-design.tsv", stringsAsFactors = FALSE)
colnames(metadata)
##  [1] "Run"                                               
##  [2] "Sample.Characteristic.age."                        
##  [3] "Sample.Characteristic.Ontology.Term.age."          
##  [4] "Sample.Characteristic.ecotype."                    
##  [5] "Sample.Characteristic.Ontology.Term.ecotype."      
##  [6] "Sample.Characteristic.organism."                   
##  [7] "Sample.Characteristic.Ontology.Term.organism."     
##  [8] "Sample.Characteristic.organism.part."              
##  [9] "Sample.Characteristic.Ontology.Term.organism.part."
## [10] "Factor.Value.environmental.stress."                
## [11] "Factor.Value.Ontology.Term.environmental.stress."  
## [12] "Analysed"
# Extract condition factor from metadata
condition <- factor(metadata[["Factor.Value.environmental.stress."]],
                    levels = c("none", "heat stress", "salt stress", "salt and heat stress"),
                    labels = c("control", "heat", "salt", "salt_heat"))
# Reorder columns in count matrix to match metadata
counts <- count_matrix[, metadata$Run]
# Ensure sample order matches
counts <- counts[, metadata$Run]
length(condition) == ncol(counts)  # Should return TRUE
## [1] TRUE
# Create DGEList object
dge <- DGEList(counts = counts, group = condition)

1.2 Filtering and normalization

# Filtering
keep <- filterByExpr(dge)
table(keep)
## keep
## FALSE  TRUE 
## 13850 18983
dge <- dge[keep, , keep.lib.sizes = FALSE] # Filter low-expression genes
nrow(dge$counts) # check retained genes after filtering
## [1] 18983
dge <- normLibSizes(dge)    # Recalculate library sizes
# Normalize for composition bias
dge <- calcNormFactors(dge, method = "TMM")

1.3 Visualize sample relationship

plotMDS(dge, col = as.numeric(dge$samples$group))
legend("bottom", inset = 0.02,
       legend = levels(dge$samples$group),
       col = 1:length(levels(dge$samples$group)),
       pch = 16,       # filled circles
       ncol = 2,       # compact layout
       horiz = FALSE,  # stacked rows
       bty = "n")

1.4 Differential gene expression analysis

#Design matrix
design <- model.matrix(~ 0 + condition)
colnames(design) <- levels(condition)
#Estimate dispersion
dge <- estimateDisp(dge, design)
plotBCV(dge)

#Check design matrix
colnames(design)
## [1] "control"   "heat"      "salt"      "salt_heat"
# Define multiple contrasts
contrasts <- makeContrasts(
  heat_vs_control = heat - control,
  salt_vs_control = salt - control,
  heat_salt_vs_control = salt_heat - control,
  levels = design)
#Fit the Quasi-Likelihood Model
fit <- glmQLFit(dge, design)
#Set up threshold for DEGs (absolute logFC ≥ 1) 
tr_heat <- glmTreat(fit, contrast = contrasts[,"heat_vs_control"], lfc = log2(2)) 

#Classify and visualize
is.de_heat <- decideTests(tr_heat)
summary(is.de_heat)
##        -1*control 1*heat
## Down                1295
## NotSig             16518
## Up                  1170
plotMD(tr_heat, status = is.de_heat,
       main = "MA Plot: Heat vs Control",
       xlab = "Average logCPM", ylab = "log2 Fold Change")

# Extract data table with DEGs
res_heat <- topTags(tr_heat, n = Inf)$table
head(res_heat)

1.5 Run one comparison at a time

#Comparison salt vs. control
tr_salt <- glmTreat(fit, contrast = contrasts[,"salt_vs_control"], lfc = log2(2))
res_salt <- topTags(tr_salt, n = Inf)$table
write.csv(res_salt, "DE_res_salt.csv")

#Comparison heat_salt vs. control
tr_heat_salt <- glmTreat(fit, contrast = contrasts[,"heat_salt_vs_control"], lfc = log2(2))
res_heat_salt <- topTags(tr_salt, n = Inf)$table
write.csv(res_heat_salt, "DE_res_heat_salt.csv")

2 Gene Set Enrichment Analysis with ClusterProfiler

The analysis performed accodring to GSEA Tutorial.

2.1 Load the packages

library(pathview)
library(clusterProfiler)
library(enrichplot)
library(ggplot2)

2.2 Set the desired organism and prepare input

library(org.At.tair.db)
organism <- "org.At.tair.db"
if (!requireNamespace(organism, quietly = TRUE)) {
  BiocManager::install(organism, character.only = TRUE)}
library(organism, character.only = TRUE)
# Load saved data DE_res_heat, provide column name X for gene IDs
df = read.csv("DE_res_heat.csv", header=TRUE)
# We want the log2 fold change 
original_gene_list <- df$logFC
# Name the vector
names(original_gene_list) <- df$X
# Omit any NA values 
gene_list<-na.omit(original_gene_list)
# Check the gene_list is named numeric vector
str(gene_list)
##  Named num [1:18983] 9.09 9.24 7.44 -5.48 -3.4 ...
##  - attr(*, "names")= chr [1:18983] "AT3G46230" "AT4G27670" "AT2G29500" "AT2G26560" ...
# Filter out non-nuclear genes
gene_list <- gene_list[grep("^AT[1-5]G", names(gene_list))]
# Sort the list in decreasing order (required for clusterProfiler)
gene_list = sort(gene_list, decreasing = TRUE)

2.3 Generate GSE list

# Check which options are available with the keytypes command
keytypes(org.At.tair.db)
##  [1] "ARACYC"       "ARACYCENZYME" "ENTREZID"     "ENZYME"       "EVIDENCE"    
##  [6] "EVIDENCEALL"  "GENENAME"     "GO"           "GOALL"        "ONTOLOGY"    
## [11] "ONTOLOGYALL"  "PATH"         "PMID"         "REFSEQ"       "SYMBOL"      
## [16] "TAIR"
# GSE list
gse <- gseGO(geneList=gene_list, 
             ont ="ALL", 
             keyType = "TAIR", 
             nPerm = 10000, 
             minGSSize = 3, 
             maxGSSize = 800, 
             pvalueCutoff = 0.05, 
             verbose = TRUE, 
             OrgDb = organism, 
             pAdjustMethod = "none")

2.4 Dot plot

library(stringr) # needed to wrap the text
